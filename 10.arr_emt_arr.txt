function emptyArrayElements() {
  let arrayInput = document.getElementById("arrayInput").value;
  let arr = arrayInput.split(',').map(item => item.trim());

  arr.length = 0;

  document.getElementById("result").textContent = "Array after emptying: [" + arr + "]";
}
