function checkArrayElement() {
  let arr = document.getElementById("arrayInput").value.split(",");
  let element = document.getElementById("checkInput").value.trim();

  // Trim spaces around each array element
  arr = arr.map(item => item.trim());

  let index = arr.indexOf(element);

  if (index !== -1) {
    document.getElementById("result").innerHTML = "Element '" + element + "' found at index " + index;
  } else {
    document.getElementById("result").innerHTML = "Element Not Found!";
  }
}
