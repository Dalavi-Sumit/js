function checkPalindrome() {     let str = document.getElementById("inputString").value;     let reversedStr = str.split("").reverse().join("");     let isPalindrome = (str === reversedStr);      if (isPalindrome) {        document.getElementById("result").innerText = `"${str}" is a Palindrome`; 
   } else { 
        document.getElementById("result").innerText = `"${str}" is NOT a Palindrome`; 
    } 
}
