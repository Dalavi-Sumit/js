function replaceText() {
    let str = document.getElementById("inputString").value;    
    let match = document.getElementById("matchText").value;    
    let pattern = document.getElementById("patternText").value;     
    let newStr = str.replace(match, pattern);     
    document.getElementById("result").innerText = "New String: " + newStr; 
}
