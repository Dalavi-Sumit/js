function recursiveReverse(str) { 
    if (str === "") {         
        return ""; 
    } 
    return recursiveReverse(str.substring(1)) + str.charAt(0); 
}; // ‚Üê added semicolon here

function reverseString() { 
    let str = document.getElementById("inputString").value;     
    let length = str.length;     
    console.log("Length of string:", length); // Optional: for debugging     
    let reversedStr = recursiveReverse(str);     
    document.getElementById("result").innerText = "Reversed String: " + reversedStr; 
}
